def generate_synthetic_video(name, duration_s, fps, resolution, pattern='circle'): 
343def generate_synthetic_video(name, duration_s, fps, resolution, pattern="circle"):
344    import cv2
345    import numpy as np
346    from pathlib import Path
347    out_path = Path(name)
348    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
349    out = cv2.VideoWriter(str(out_path), fourcc, fps, resolution)
350    w, h = resolution
351    for t in range(int(duration_s * fps)):
352        frame = np.zeros((h, w, 3), dtype=np.uint8)
353        if pattern == "circle":
354            center = (int(w//2 + 100*np.sin(t*0.1)), int(h//2 + 50*np.cos(t*0.2)))
355            cv2.circle(frame, center, 30, (0,255,0), -1)
356        elif pattern == "color_switch":
357            color = [(255,0,0), (0,255,0), (0,0,255)][(t//30) % 3]
358            frame[:] = color
359        elif pattern == "noise":
360            frame = np.random.randint(0, 50, (h, w, 3), dtype=np.uint8) + 100
361        out.write(frame)
362    out.release()
363    return out_path
